# 03-15. Несчастливые билеты — брутфорс
## Условие tl;dr
Даны:
- N — номер билета
- K — основание системы счисления
Нужно найти минимальное число X ≥ N, такое что в записи числа X по основанию K:
- сумма цифр левой половины равна сумме цифр правой половины
- если длина нечётная, слева добавляется ведущий ноль

## Решение — псевдокод

``` rust
fn is_lucky(a: u32, k: u32) -> bool {
    // b нужен, чтобы не портить исходное a,
    // пока считаем сумму всех цифр и длину
    let b = a;

    let sum_all = 0;
    let cnt = 0;

    // Считаем:
    // 1) сумму всех цифр
    // 2) количество цифр
    while b > 0 {
        sum_all += b % k;
        cnt += 1;
        b /= k;
    }

    // Отдельный случай: число 0
    if cnt == 0 {
        return true;
    }

    // Если длина нечётная, добавляем ведущий 0
    if cnt % 2 == 1 {
        cnt += 1;
    }

    // Считаем сумму правой половины.
    // Так как идём с конца числа, берём младшие цифры.
    let sum_right = 0;

    // После выравнивания длины до 2m
    // цикл выполнится ровно m раз
    while cnt > 0 {
        sum_right += a % k;
        a /= k;
        cnt -= 2;
    }

    // left + right = sum_all
    // left == right  <=>  2 * right == sum_all
    return sum_right * 2 == sum_all;
}

fn solve() {
    read n, k;

    let x = n;

    // Перебираем числа подряд,
    // пока не найдём первое счастливое
    while !is_lucky(x, k) {
        x += 1;
    }

    print x;
}
```
